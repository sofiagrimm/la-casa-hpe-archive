// ===== ADDITIONAL EDITING PRIVILEGES ===== // Duplicate/Clone Archive window.duplicateArchive = async function(id) { try { const { data: original, error } = await sb.from('archives').select('*').eq('id', id).single(); if (error) throw error; // Remove id and created_at, mark as draft const duplicate = { ...original }; delete duplicate.id; delete duplicate.created_at; duplicate.title = `${original.title} (Copy)`; duplicate.status = 'pending'; const { error: insertError } = await sb.from('archives').insert([duplicate]); if (insertError) throw insertError; showToast('Archive duplicated successfully!', 'success'); loadDashboard(); } catch (error) { console.error('Duplicate error:', error); showToast('Failed to duplicate archive', 'error'); } }; // Export Single Archive as JSON window.exportArchive = async function(id) { try { const { data, error } = await sb.from('archives').select('*').eq('id', id).single(); if (error) throw error; const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = `archive_${id}_${Date.now()}.json`; a.click(); URL.revokeObjectURL(url); showToast('Archive exported!', 'success'); } catch (error) { console.error('Export error:', error); showToast('Failed to export archive', 'error'); } }; // Export All Archives as CSV window.exportAllArchives = async function() { try { const { data, error } = await sb.from('archives').select('*').order('created_at', { ascending: false }); if (error) throw error; // Convert to CSV const headers = Object.keys(data[0] || {}); const csv = [ headers.join(','), ...data.map(row => headers.map(h => JSON.stringify(row[h] || '')).join(',')) ].join('\n'); const blob = new Blob([csv], { type: 'text/csv' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = `all_archives_${Date.now()}.csv`; a.click(); URL.revokeObjectURL(url); showToast(`Exported ${data.length} archives!`, 'success'); } catch (error) { console.error('Export all error:', error); showToast('Failed to export archives', 'error'); } }; // Bulk Edit - Change status or type for selected archives window.bulkEdit = async function() { const checkedBoxes = document.querySelectorAll('.archive-checkbox:checked'); const ids = Array.from(checkedBoxes).map(cb => cb.dataset.id); if (ids.length === 0) { showToast('No archives selected', 'error'); return; } const action = prompt('Choose action:\n1. Change Status (pending/published)\n2. Change Media Type\nEnter 1 or 2:'); if (action === '1') { const status = prompt('Enter new status (pending or published):'); if (!status || !['pending', 'published'].includes(status)) return; try { for (const id of ids) { await sb.from('archives').update({ status }).eq('id', id); } showToast(`Updated ${ids.length} archive(s)`, 'success'); loadDashboard(); } catch (error) { console.error('Bulk edit error:', error); showToast('Failed to update archives', 'error'); } } else if (action === '2') { const type = prompt('Enter media type (photo/video/audio/document/flyer):'); if (!type) return; try { for (const id of ids) { await sb.from('archives').update({ media_type: type }).eq('id', id); } showToast(`Updated ${ids.length} archive(s)`, 'success'); loadDashboard(); } catch (error) { console.error('Bulk edit error:', error); showToast('Failed to update archives', 'error'); } } };
